#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

int add(int x, int y);
int minus(int x, int y);
int murti(int x, int y);
int devide(int x, int y);
int s_murti(int x, int y);


int main() {
	int select = 0;
	int x, y = 0;
	while (1) {
		scanf("%d", &select);
		if (select == 1) {
			scanf("%d %d", &x, &y);
			printf("\n%d\n", add(x, y));
		}
		else if (select == 2) {
			scanf("%d %d", &x, &y);
			printf("\n%d\n", minus(x, y));
		}
		else if (select == 3) {
			scanf("%d %d", &x, &y);
			printf("\n%d\n", murti(x, y));
		}
		else if (select == 4) {
			scanf("%d %d", &x, &y);
			printf("\n%d\n", devide(x, y));
		}
		else if (select == 5) {
			scanf("%d %d", &x, &y);
			printf("\n%d\n", s_murti(x, y));
		}
	}

	return 0;
}
int add(int x, int y) {
	return x + y;
}
int minus(int x, int y) {
	return x - y;
}
int murti(int x, int y) {
	return x * y;
}
int devide(int x, int y) {
	return x / y;
}
int s_murti(int x, int y) {
	int z = 1;
	for (int i = 0; i < y; i++) {
		z = z * x;
	}
	return z;

}
