void draw_board(int x1, int y1, int x2, int y2) {
	system("cls");
	for (int y = 1; y < 11; y++) {
		for (int x = 1; x < 11; x++) {
			if (x1 == x) {
				if (y1 == y) {
					printf("@  ");
				}
				else if (x2 == x) {
					if (y2 == y) {
						printf("$  ");
					}
					else {
						printf("*  ");
					}
				}
				else {
					printf("*  ");
				}
			}
			else if (x2 == x) {
				if (y2 == y) {
					printf("$  ");
				}
				else {
					printf("*  ");
				}
			}
			else {
				printf("*  ");
			}

		}
		printf("\n");

	}
}

void gotoxy(int x, int y) {
	COORD pos = { x, y };
	SetConsoleCursorPosition(GetStdHandle(STD_OUTPUT_HANDLE), pos);
}
