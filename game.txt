#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>

int main() {
	int select = 0;
	int x1 = 1; 
	int y1 = 1;
	int x2 = 10;
	int y2 = 10;
	while (1) {
		printf("\n\n");
		for (int y = 1; y < 11; y++) {
			for (int x = 1; x < 11; x++) {
				if (x1 == x) {
					if (y1 == y) {
						printf("@  ");
					}
					else if (x2 == x) {
						if (y2 == y) {
							printf("$  ");
						}
						else {
							printf("*  ");
						}
					}
					else {
						printf("*  ");
					}
				}
				else if (x2 == x) {
					if (y2 == y) {
						printf("$  ");
					}
					else {
						printf("*  ");
					}
				}
				else {
					printf("*  ");
				}
				
			}
			printf("\n");
		}
		scanf("%d", &select);
		if (select == 1) {
			y1--;
		}
		else if (select == 2) {
			y1++;
		}
		else if (select == 3) {
			x1--;
		}
		else if (select == 4) {
			x1++;
		}
	}
	return 0;
}
